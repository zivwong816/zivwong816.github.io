package main

import (
   "net/http"
   "log"
   "fmt"
   "time"
)

func printUrl(url string)  {
   resp, err := http.Get(url)
   if err != nil {
      log.Fatal(err)
      return
   }
   defer resp.Body.Close()
   fmt.Println(url, resp.Status)
}
// 主协程启动一个work协程，同时传递一个channelo
// 主协程向channel里面发送一个url
// work协程启出channel里面获取url，之后调用printUrl
func work(ch chan string)  {
   url := <-ch
   printUrl(url)
}

func main()  {
   ch := make(chan string)
   go work(ch)
   url := "http://www.qq.com"
   ch <-url
   time.Sleep(3 * time.Second)
}

package main

import (
   "net/http"
   "log"
   "fmt"
   "time"
)

func printUrl(url string)  {
   resp, err := http.Get(url)
   if err != nil {
      log.Fatal(err)
      return
   }
   defer resp.Body.Close()
   fmt.Println(url, resp.Status)
}

//1 只要不close可以永远发送数据和接受数据
//2 如果channel里面没有数据，接收方阻塞
//3 如果没有人正在等待channel的数据，发送方会阻塞
//4 从一个close的channel取数据永远不会阻塞，同时获取的是默认值
//默认值是chen类型，如果是字符串就是空


// 主协程启动3个work协程
// 主协程向channel里面发送多个url，发送完毕之后关闭channel
// work协程从channel里面获取url，之后调用printUrl打印url
// work协程不停重复第三条，直到channel关闭
func work(ch chan string)  {
   for url := range ch {
      printUrl(url)
   }
}

func main()  {
   ch := make(chan string)
   for i :=0 ;i < 3;i++ {
      go work(ch)
   }
   urls := []string{"http://www.qq.com","http://www.baidu.com"}
   for _, url := range urls {
      ch <- url
   }
   close(ch)
   time.Sleep(3 * time.Second)
}

package main

import (
   "net/http"
   "log"
   "fmt"
   "sync"
)

func printUrl(url string)  {
   resp, err := http.Get(url)
   if err != nil {
      log.Fatal(err)
      return
   }
   defer resp.Body.Close()
   fmt.Println(url, resp.Status)
}

// 创建一个waitGroup
// 调用add
// 调用wait等待work协程结束
func work(ch chan string, wg *sync.WaitGroup)  {
   for url := range ch {
      printUrl(url)
   }
   wg.Done()

func main()  {
   ch := make(chan string)
   wg := new(sync.WaitGroup)
   //wg.Add(3)
   for i :=0 ;i < 3;i++ {
      wg.Add(1)
      go work(ch, wg)
   }
   urls := []string{"http://www.qq.com","http://www.baidu.com"}
   for _, url := range urls {
      ch <- url
   }
   close(ch)
   wg.Wait()
}
